await chatDataService.init();
const db = await chatDataService.getDatabase();

window.sql = (query, params = []) => {
  const stmt = db.prepare(query);
  try {
    stmt.bind(params);
    const rows = [];
    while (stmt.step()) rows.push(stmt.getAsObject());
    return rows;
  } finally {
    stmt.free();
  }
};
